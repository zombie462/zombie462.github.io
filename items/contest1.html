#include <bits/stdc++.h>
using namespace std;
int read(){
	int x=0,f=1;
	char ch=getchar();
	while (ch<'0' || ch>'9'){
		if (ch=='-') f=-f;
		ch=getchar();
	}
	while (ch>='0' && ch<='9'){
		x=x*10+ch-48;
		ch=getchar();
	}
	return x*f;
}
const int mod=1000000007;
const int N=200010;
int a[N];
int main(){
	int n=read(),k=read();
	for (int i=1;i<=n;++i) a[i]=read();
	int ans=0;
	for (int i=1;i<=n;++i){
		int sum=0,rpos=i;
		for (int j=i+1;j<=n;++j){
			if (sum>k) break;
			rpos=j;
			if (a[j]>a[i]) sum++;
		}
		if (sum>k) rpos--,sum--;
		for (int j=i;j>=1;--j){
			if (a[j]>=a[i]){
				sum++;
				if (sum>k){
					while (rpos>i && a[rpos]<=a[i]) rpos--;
					if (rpos<=i) break;
					rpos--;sum--;
				}
			}
			ans+=1LL*a[i]*(rpos-i+1)%mod;ans%=mod;
		}
	}
	cout<<ans<<endl;
	return 0;
}
