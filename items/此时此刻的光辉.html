实现步骤：

1. 分解序列中的每个数
        对于每个数，除掉它所有1000以内的质因子，然后丢进前缀和数组。然后这个数显然只能最多分解两个质数相乘。用PR找到这两个质数，离散化掉即可。
2. 开始莫队
        设g[i]表示离散化后id为i的质因子的个数。移动莫队指针只要将答案乘上g[i]/(g[i]+1)或者(g[i]+1)/g[i]，并修改g[i]。最后处理那些小于1000的质因子即可。

#include <bits/stdc++.h>
using namespace std;
#define int long long
const int p=19260817;
#define N 111111
#define M 222
#define G 1111
int tot,pr[M],inv[N*2];
bool f[G];
int a[N],b[N],s[M][N],ans[N],g[N<<1],res=1;
struct pt{
    int l,r,b,id;
    bool operator < (pt a) const {
        return b<a.b || (b==a.b && ((b&1)?r<a.r:r>a.r));
    }
}q[N]; 
int read(){
    int x=0,f=1;
    char ch=getchar();
    while (ch<'0' || ch>'9'){
        if (ch=='-') f=-1;
        ch=getchar();
    }
    while (ch>='0' && ch<='9'){
        x=x*10+ch-'0';ch=getchar();
    }
    return x*f;
}
unordered_map <int,int> mp;
void getprime(int n){
    f[1]=1;
    int k=0;
    for (int i=2;i<=n;++i){
        if (!f[i]) pr[++tot]=i;
        for (int j=1;k=i*pr[j],j<=tot && k<=n;++j){
            f[k]=1;
            if (!(i%pr[j])) break;
        }
    }
}
void getinv(int n){
    inv[1]=1;
    for (int i=2;i<=n;++i) inv[i]=(p-p/i)*inv[p%i]%p;
}
int gcd(int x,int y){
    if (x%y==0) return y;
    else return gcd(y,x%y);
}
int fastpow(int x,int y,int p){
    int res=1;
    while (y){
        if (y&1) res=res*1LL*x%p;
        x=x*1LL*x%p;
        y>>=1;
    }
    return res;
}
bool miller_rabin(int n){
    if (n<=1000) return f[n];
    int m=n-1,r=m,t=0;
    while (!(r&1)){
        r>>=1;t++;
    }
    for (int i=1;i<=10;++i){
        int x=fastpow(pr[i],r,n);
        for (int j=0;j<t && x>1;++j){
            int y=x*1LL*x%n;
            if (y==1 && x!=m) return 1;
            x=y;
        }
        if (x!=1) return 1;
    }
    return 0;
}
int pollard_rho(int n){
    int s=0,t=0,c=rand()%(n-1)+1,w=1,d=0,g=1;
    while (true){
        for (int p=0;p<g;++p){
            t=(t*1LL*t+c)%n;
            w=w*1LL*(t>s?(t-s):(s-t))%n;
            if (!(p&63)){
                d=gcd(w,n);
                if (d>1) return d;
            } 
        }
        d=gcd(w,n);
        if (d>1) return d;
        g<<=1;s=t;w=1;
    }
}
void inc(int x){
    if (a[x]){
        res=res*1LL*inv[g[a[x]]]%p*(g[a[x]]+1)%p;
        g[a[x]]++;
        if (b[x]){
            res=res*1LL*inv[g[b[x]]]%p*(g[b[x]]+1)%p;
            g[b[x]]++;
        }
    }
}
void dec(int x){
    if (a[x]){
        res=res*1LL*inv[g[a[x]]]%p*(g[a[x]]-1)%p;
        g[a[x]]--; 
        if (b[x]){
            res=res*1LL*inv[g[b[x]]]%p*(g[b[x]]-1)%p;
            g[b[x]]--;
        }
    } 
}
signed main(){
    int n=read(),m=read();
    int blen=sqrt(n);
    srand((unsigned)time(NULL));
    getprime(1000);
    getinv(n<<1);
    int id=0;
    for (int i=1;i<=n;++i){
        a[i]=read();
        for (int j=1;j<=tot;++j){
            while (!(a[i]%pr[j])){
                a[i]/=pr[j];s[j][i]++;
            }
            s[j][i]+=s[j][i-1];
        }
        if (a[i]>1 && miller_rabin(a[i])){
            b[i]=a[i];
            while (b[i]==a[i]) b[i]=pollard_rho(a[i]);
            a[i]/=b[i];
        }
        if (a[i]>1){
            if (b[i]){
                if (!mp.count(b[i])) mp[b[i]]=++id;
                b[i]=mp[b[i]];
            }
            if (!mp.count(a[i])) mp[a[i]]=++id;
            a[i]=mp[a[i]];
        }else a[i]=0;
    }
    for (int i=1;i<=id;++i) g[i]=1;
    for (int i=1;i<=m;++i){
        q[i].l=read();q[i].r=read();
        q[i].b=q[i].l/blen;
        q[i].id=i;
    }
    sort(q+1,q+1+m);
    int l=1,r=0;
    for (int i=1;i<=m;++i){
        while (l>q[i].l) inc(--l);
        while (r<q[i].r) inc(++r);
        while (l<q[i].l) dec(l++);
        while (r>q[i].r) dec(r--);
        int sum=res;
        for (int j=1;j<=tot;++j){
            sum=sum*1LL*(s[j][r]-s[j][l-1]+1)%p;
        }
        ans[q[i].id]=sum;
    }
    for (int i=1;i<=m;++i) printf("%d\n",ans[i]);
    return 0;
}
